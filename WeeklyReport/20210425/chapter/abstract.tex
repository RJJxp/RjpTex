\renewcommand\abstractname{\Large\textbf{摘要}}
\begin{abstract}
    \center{
    \parbox[s]{12cm}{\qquad 三维激光扫描技术是一种先进的全自动高精度立体扫描技术, 通过三维激光扫描仪获取目标物表面各点的空间坐标, 得到点云数据.然后由获取的测量数据构造出目标物的三维模型的一种全自动测量技术. 其中, 平面拟合是地面三维激光扫面数据处理的常用方法之一. 本篇课程报告对平面拟合的方法做了简单介绍并对特征根拟合方法进行了实现. 使用 python 编程语言, numpy, sklearn 库用于计算, matplotlib 库用于可视化}
    }
    \\[12pt]
    \par\textbf{关键字:} 激光点云, 平面拟合, 可视化

\end{abstract}
